vue环境

# vue 环境

# demo

# 打包环境判断
NODE_ENV=production
# 项目内环境判断
VUE_APP_ENV=production

# 打包路径及配置路径
VUE_APP_BASE_URL=/h5/

# 域名
VUE_APP_SERVICE_URL=https://baidu.com

# 是否去掉控制台打印
VUE_APP_CONSOLE=true

# 是否开启调试模式 vconsole
VUE_APP_VCONSOLE=false

# 是否开启加密模式
VUE_APP_ENCRYPT=true

# CDN
VUE_APP_CDN='https://cdn.com'
