  1. 不包含重复的数字
function permute(nums) {
  let all = [];
  // let nums = [1, 2, 3];
  function loop(path, nums) {
    if (path.length == nums.length) {
      all.push(path.slice(0));
      return;
    }
    for (let i = 0; i < nums.length; i++) {
      if (path.includes(nums[i])) continue;
      path.push(nums[i]);
      loop(path);
      path.pop();
    }
  }
  loop([], nums);

  return all;
}
